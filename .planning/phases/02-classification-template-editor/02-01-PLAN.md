```typescript
import { Router } from 'express';
import path from 'node:path';
import fs from 'node:fs/promises';
import { z } from 'zod';
import { assertSafePath } from '../middleware/security.js';
import type { TemplateName } from '../../shared/types/templates.js';

const router = Router();

const REPO_ROOT = process.env.ENTRAOPS_ROOT ?? path.resolve(import.meta.dirname, '../../..');
const TEMPLATES_BASE = path.join(REPO_ROOT, 'Classification', 'Templates');
const GLOBAL_PATH = path.join(REPO_ROOT, 'Classification', 'Global.json');

const safeTemplatePath = assertSafePath(TEMPLATES_BASE);
```

**Zod schemas (Zod v4 — use `error:` not `message:` for custom errors):**

```typescript
const TierLevelDefinitionEntrySchema = z.object({
  Category: z.string().min(1),
  Service: z.string().min(1),
  RoleAssignmentScopeName: z.array(z.string()),
  RoleDefinitionActions: z.array(z.string()),
});

const TierBlockSchema = z.object({
  EAMTierLevelName: z.enum(['ControlPlane', 'ManagementPlane', 'UserAccess']),
  EAMTierLevelTagValue: z.string(),
  TierLevelDefinition: z.array(TierLevelDefinitionEntrySchema),
});

const TemplateFileSchema = z.array(TierBlockSchema);

const GlobalFileSchema = z.array(
  z.object({ ExcludedPrincipalId: z.array(z.string()) })
);
```

**Helper: strip BOM + parse JSON:**
```typescript
function parseBomJson(raw: string): unknown {
  return JSON.parse(raw.replace(/^\uFEFF/, ''));
}
```

**Helper: atomic write (temp file → rename):**
```typescript
async function atomicWrite(filePath: string, content: string): Promise<void> {
  const tmp = filePath + '.tmp';
  await fs.writeFile(tmp, content, 'utf-8');
  await fs.rename(tmp, filePath);
}
```

**Route: GET /api/templates (list)**
Returns `{ names: TemplateName[] }`.

**Route: GET /api/templates/global**
Reads GLOBAL_PATH (no path traversal check needed — fixed path). Parses JSON, strips BOM. Returns `{ exclusions: parsed[0].ExcludedPrincipalId }` (empty array if file missing or ExcludedPrincipalId empty).

**Route: PUT /api/templates/global**
Body: `{ exclusions: string[] }`. Validate each as UUID via `z.array(z.string().uuid())`. If invalid, return 400 with `{ error: result.error.flatten() }`. If valid: wrap as `[{ ExcludedPrincipalId: data.exclusions }]`, atomic write to GLOBAL_PATH with JSON.stringify(…, null, 2). Return 200 `{ ok: true }`.

**Route: GET /api/templates/:name**
Validate `:name` is one of the 5 template names (return 400 if not). Build file path: `safeTemplatePath(name + '.json')` — this throws 403 on traversal. Read file, parseBomJson, validate via TemplateFileSchema (return 400 if fails). Return `{ name, tiers: parsed }`.

**Route: PUT /api/templates/:name**
Validate `:name` is one of the 5. Body: `{ tiers: TierBlock[] }`. Run `TemplateFileSchema.safeParse(req.body.tiers)` — return 400 on failure with `{ error: result.error.flatten() }`. Build safe path. Atomic write `JSON.stringify(req.body.tiers, null, 2)`. Return 200 `{ ok: true }`.

**Important:** Register `/global` route BEFORE `/:name` so "global" isn't captured by the param.
