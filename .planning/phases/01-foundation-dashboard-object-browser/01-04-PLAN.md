PrivilegedEAM file layout:
- ENTRAOPS_ROOT env var points to repo root (parent of PrivilegedEAM/)
- Aggregate files: {ENTRAOPS_ROOT}/PrivilegedEAM/{RbacSystem}/{RbacSystem}.json
- Per-object files: {ENTRAOPS_ROOT}/PrivilegedEAM/{RbacSystem}/{ObjectId}.json
- RBAC systems: EntraID, ResourceApps, IdentityGovernance, DeviceManagement, Defender

Zod v4 API differences from v3 (anti-patterns to avoid):
- z.record(z.string()) → z.record(z.string(), z.string())
- { message: 'Too short' } → { error: 'Too short' }
- z.object({}).strict() → z.strictObject({})

Express v5 patterns:
- Async route handlers auto-forward thrown errors to 4-arg errorHandler (no try/catch needed)
- res.json(data, 200) removed — use res.status(200).json(data) or just res.json(data)
- z.array() in Zod for repeated query params but Express parses them as string | string[]
  Use z.preprocess to handle single string → array: z.preprocess(v => [v].flat(), z.array(z.string()).optional())

```typescript
import { Router } from 'express';
import fs from 'node:fs/promises';
import path from 'node:path';
import { readEamJson } from '../services/eamReader.js';
import { getRecentPrivilegedEAMCommits } from '../services/gitLog.js';
import type { PrivilegedObject, EamTier, RbacSystem } from '../../shared/types/eam.js';
import type { DashboardResponse, TierPimCounts, RbacBreakdown } from '../../shared/types/api.js';

const router = Router();

// ENTRAOPS_ROOT: repo root dir (parent of PrivilegedEAM/)
const REPO_ROOT = process.env.ENTRAOPS_ROOT ?? path.resolve('../../..');
const EAM_BASE = path.join(REPO_ROOT, 'PrivilegedEAM');
const RBAC_SYSTEMS: RbacSystem[] = ['EntraID', 'ResourceApps', 'IdentityGovernance', 'DeviceManagement', 'Defender'];

async function getEamFreshness(): Promise<string | null> {
  const mtimes: number[] = [];
  for (const system of RBAC_SYSTEMS) {
    try {
      const aggregateFile = path.join(EAM_BASE, system, `${system}.json`);
      const stat = await fs.stat(aggregateFile);
      mtimes.push(stat.mtimeMs);
    } catch {
      // File not present — skip
    }
  }
  if (mtimes.length === 0) return null;
  return new Date(Math.max(...mtimes)).toISOString();
}

router.get('/', async (_req, res) => {
  // Per-widget error isolation: collect all objects, handle per-system errors
  const allObjects: PrivilegedObject[] = [];
  let hasData = false;

  for (const system of RBAC_SYSTEMS) {
    try {
      const aggregateFile = path.join(EAM_BASE, system, `${system}.json`);
      const objects = await readEamJson(aggregateFile) as PrivilegedObject[];
      if (Array.isArray(objects) && objects.length > 0) {
        allObjects.push(...objects);
        hasData = true;
      }
    } catch {
      // Per CONTEXT.md: per-widget isolation — one broken file fails its card only
      // The route still returns; UI handles missing data per-system
    }
  }

  if (!hasData) {
    const response: DashboardResponse = {
      hasData: false,
      tiers: { ControlPlane: 0, ManagementPlane: 0, UserAccess: 0, Unclassified: 0 },
      objectTypes: {},
      rbacBreakdown: [],
      pimTypes: {
        ControlPlane: { Permanent: 0, Eligible: 0 },
        ManagementPlane: { Permanent: 0, Eligible: 0 },
        UserAccess: { Permanent: 0, Eligible: 0 },
      },
      freshness: null,
      recentCommits: [],
    };
    res.json(response);
    return;
  }

  // Tier counts (DASH-01)
  const tiers = { ControlPlane: 0, ManagementPlane: 0, UserAccess: 0, Unclassified: 0 };
  // Object type breakdown (DASH-02)
  const objectTypes: Record<string, number> = {};
  // PIM assignment type (DASH-04) — counts per tier
  const pimTypes: TierPimCounts = {
    ControlPlane: { Permanent: 0, Eligible: 0 },
    ManagementPlane: { Permanent: 0, Eligible: 0 },
    UserAccess: { Permanent: 0, Eligible: 0 },
  };
  // RBAC breakdown per tier (DASH-03)
  const rbacByTier = new Map<EamTier, Record<RbacSystem, number>>();

  for (const obj of allObjects) {
    const tier = obj.ObjectAdminTierLevelName;
    tiers[tier] = (tiers[tier] ?? 0) + 1;
    objectTypes[obj.ObjectType] = (objectTypes[obj.ObjectType] ?? 0) + 1;

    // RBAC breakdown
    if (!rbacByTier.has(tier)) {
      rbacByTier.set(tier, { EntraID: 0, ResourceApps: 0, IdentityGovernance: 0, DeviceManagement: 0, Defender: 0 });
    }
    const rbacRow = rbacByTier.get(tier)!;
    rbacRow[obj.RoleSystem] = (rbacRow[obj.RoleSystem] ?? 0) + 1;

    // PIM type counts by tier
    if (tier in pimTypes) {
      for (const assignment of obj.RoleAssignments ?? []) {
        if (assignment.PIMAssignmentType === 'Permanent') {
          pimTypes[tier as keyof TierPimCounts].Permanent += 1;
        } else if (assignment.PIMAssignmentType === 'Eligible') {
          pimTypes[tier as keyof TierPimCounts].Eligible += 1;
        }
      }
    }
  }

  const rbacBreakdown: RbacBreakdown[] = [];
  for (const [tier, counts] of rbacByTier.entries()) {
    rbacBreakdown.push({ tier, ...counts });
  }

  const [freshness, recentCommits] = await Promise.all([
    getEamFreshness(),
    getRecentPrivilegedEAMCommits(5),
  ]);

  const response: DashboardResponse = {
    hasData: true,
    tiers,
    objectTypes,
    rbacBreakdown,
    pimTypes,
    freshness,
    recentCommits,
  };

  res.json(response);
});

export { router as dashboardRouter };
```

```typescript
import { Router } from 'express';
import fs from 'node:fs/promises';
import path from 'node:path';
import { z } from 'zod';
import { readEamJson } from '../services/eamReader.js';
import type { PrivilegedObject, EamTier, RbacSystem, ObjectType, PimAssignmentType } from '../../shared/types/eam.js';

const router = Router();

const REPO_ROOT = process.env.ENTRAOPS_ROOT ?? path.resolve('../../..');
const EAM_BASE = path.join(REPO_ROOT, 'PrivilegedEAM');
const RBAC_SYSTEMS: RbacSystem[] = ['EntraID', 'ResourceApps', 'IdentityGovernance', 'DeviceManagement', 'Defender'];

// coerceArray: Express may parse repeated params as string | string[]
// Zod v4: use z.preprocess to normalize to string[]
const coerceArray = z.preprocess(
  (v) => (v === undefined ? [] : Array.isArray(v) ? v : [v]),
  z.array(z.string())
);

// Zod v4 schema — note: { error: } not { message: }
const QuerySchema = z.object({
  tier:     coerceArray.optional(),
  rbac:     coerceArray.optional(),
  type:     coerceArray.optional(),
  pim:      coerceArray.optional(),
  onprem:   coerceArray.optional(),
  q:        z.string().optional(),
  page:     z.coerce.number({ error: 'page must be a number' }).int().min(0).default(0),
  pageSize: z.coerce.number({ error: 'pageSize must be a number' }).int().min(1).max(200).default(50),
  sortBy:   z.string().default('ObjectAdminTierLevel'),
  sortDir:  z.enum(['asc', 'desc']).default('asc'),
});

async function loadAllObjects(): Promise<PrivilegedObject[]> {
  const all: PrivilegedObject[] = [];
  for (const system of RBAC_SYSTEMS) {
    try {
      const file = path.join(EAM_BASE, system, `${system}.json`);
      const objects = await readEamJson(file) as PrivilegedObject[];
      if (Array.isArray(objects)) all.push(...objects);
    } catch {
      // Skip systems with missing/unreadable files
    }
  }
  return all;
}

// GET /api/objects — paginated, filtered, sorted list
router.get('/', async (req, res) => {
  const parsed = QuerySchema.safeParse(req.query);
  if (!parsed.success) {
    res.status(400).json({ error: parsed.error.flatten() });
    return;
  }

  const { tier, rbac, type, pim, onprem, q, page, pageSize, sortBy, sortDir } = parsed.data;
  let objects = await loadAllObjects();

  // Filtering
  if (tier && tier.length > 0) {
    objects = objects.filter(o => tier.includes(o.ObjectAdminTierLevelName));
  }
  if (rbac && rbac.length > 0) {
    objects = objects.filter(o => rbac.includes(o.RoleSystem));
  }
  if (type && type.length > 0) {
    objects = objects.filter(o => type.includes(o.ObjectType));
  }
  if (pim && pim.length > 0) {
    objects = objects.filter(o =>
      o.RoleAssignments?.some(r => pim.includes(r.PIMAssignmentType))
    );
  }
  if (onprem && onprem.length > 0) {
    objects = objects.filter(o => {
      const val = o.OnPremSynchronized;
      return onprem.includes(val === true ? 'true' : val === false ? 'false' : 'null');
    });
  }
  if (q && q.trim()) {
    const search = q.trim().toLowerCase();
    objects = objects.filter(o =>
      o.ObjectDisplayName?.toLowerCase().includes(search) ||
      o.ObjectUserPrincipalName?.toLowerCase().includes(search)
    );
  }

  // Sorting
  const sortKey = sortBy as keyof PrivilegedObject;
  objects.sort((a, b) => {
    const av = String(a[sortKey] ?? '');
    const bv = String(b[sortKey] ?? '');
    return sortDir === 'asc' ? av.localeCompare(bv) : bv.localeCompare(av);
  });

  const total = objects.length;
  const sliced = objects.slice(page * pageSize, (page + 1) * pageSize);

  res.json({ objects: sliced, total, page, pageSize });
});

// GET /api/objects/:id — single object by ObjectId
router.get('/:id', async (req, res) => {
  const { id } = req.params;
  const all = await loadAllObjects();
  const found = all.find(o => o.ObjectId === id);
  if (!found) {
    res.status(404).json({ error: 'Not found' });
    return;
  }
  res.json(found);
});

export { router as objectsRouter };
```
